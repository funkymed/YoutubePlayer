var player = $('#video').jQYTPlayer('mxfmxi-boyo',{
    width:500,
    height:360,
    showinfo:false,
    controls:false,
    autoplay:false,
    loop:true
    });
    player.onPlaying=function()
    {
    $('.progress .bar').width(this.getProgress()+'%');
    };
    player.onReadyCallback=function()
    {
    $('.title').html(this.data.title);
    };
    player.loader=$('.loader');

    $('button').on('click',function()
    {
    switch($(this).html())
    {
    case 'Play':
    player.play();
    break;
    case 'Pause':
    player.pause();
    break;
    case 'Stop':
    player.stop();
    break;
    }
    });
